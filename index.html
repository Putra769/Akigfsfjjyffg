let handler = async (m, { Thezy, args, reply }) => {
  let users = m.mentionedJid?.[0] 
    || (m.quoted ? m.quoted.sender : (args[0] ? args[0].replace(/[^0-9]/g, '') + "@s.whatsapp.net" : null));

  if (!users) return m.reply("❌ Harap tag atau reply pengguna yang ingin dikeluarkan!");

  try {
    await Thezy.groupParticipantsUpdate(m.chat, [users], "remove");
    m.reply(`✅ @${users.split("@")[0]} telah dikeluarkan dari grup!`);
  } catch (error) {
    m.reply("⚠️ Gagal mengeluarkan pengguna. Pastikan bot memiliki izin admin!");
  }
};

handler.tags = ["groupmenu"];
handler.command = ["kick"];
handler.description = ["Mengeluarkan anggota dari grup"];
handler.group = true;
handler.admin = true;
handler.botAdmin = true;

module.exports = handler;
JS • Plugin hidetag 
const handler = async (m, { Thezy, text }) => {
    if (!text) return m.reply("Teks tidak boleh kosong.");

    let groupMetadata = await Thezy.groupMetadata(m.chat);
    let tagJid = groupMetadata.participants.map(p => p.id);

    await Thezy.sendMessage(m.chat, {
        text,
        mentions: tagJid
    }, { quoted: m });
};

handler.command = ["hidetag"];
handler.tags = ["groupmenu"];
handler.description = ["Tag semua member"];
handler.group = true;
handler.admin = true;
handler.botAdmin = true;

module.exports = handler;

// Accordion Menu JavaScript
document.querySelectorAll('.accordion-button').forEach(button => {
  button.addEventListener('click', () => {
    const content = button.nextElementSibling;
    
    // Close all other accordion items
    document.querySelectorAll('.accordion-content').forEach(item => {
      if (item !== content) {
        item.classList.remove('active');
      }
    });
    
    // Toggle current item
    content.classList.toggle('active');
  });
});
